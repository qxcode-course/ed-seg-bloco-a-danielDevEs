package main

import (
	"fmt"
	"math/rand"
)

func randInt(min, max int) int {
	return min + rand.Intn(max-min+1)
}

func triangulinho(pen *Pen, lado float64){
	if lado < 1{
		return
	}
	for range 3{
		pen.Walk(lado)
		pen.Right(120)
		triangulinho(pen, lado*0.5)
	}
}

func main() {
	pen := NewPen(1500, 1500)
	pen.SetPosition(400, 550)
	pen.SetHeading(0)
	
	triangulinho(pen, 700)
	
	pen.SavePNG("tree.png")
	fmt.Println("PNG file created successfully.")
}
